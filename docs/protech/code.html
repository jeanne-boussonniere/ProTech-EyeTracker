 37def nettoyage (gaze2dx,gaze2dy,media_time_stamp, gaze_class):
 38    """
 39    L'objectif est de supprimer le bruit et les données invalides (clignements, perte de tracking).\n
 40        • Entrées : Listes brutes Gaze2dX, Gaze2dY, MediaTimeStamp, GazeClass.\n
 41        • Logique : La fonction itère sur les index. Elle conserve uniquement les points où GazeClass \n
 42                    est valide (différent de nan, None ou vide).\n
 43        • Sortie : Listes filtrées et synchronisées x_filtered, y_filtered, time_filtered.
 44    """
 45
 46    min_len = min(len(gaze2dx), len(gaze2dy), len(media_time_stamp), len(gaze_class))
 47    fixations_idx = [i for i in range(min_len) if str(gaze_class[i]).strip() not in ['nan', '', 'None']]
 48    fixation_gaze2dx = [gaze2dx[i] for i in fixations_idx]
 49    fixation_gaze2dy = [gaze2dy[i] for i in fixations_idx]
 50    fixation_time_stamp = [media_time_stamp[i] for i in fixations_idx]
 51    gaze2dx_filtered = []
 52    gaze2dy_filtered = []
 53    time_stamp_filtered = []
 54    for x, y, t in zip(fixation_gaze2dx, fixation_gaze2dy, fixation_time_stamp):
 55        try:
 56            gaze2dx_filtered.append(float(x))
 57            gaze2dy_filtered.append(float(y))
 58            time_stamp_filtered.append(float(t))
 59        except ValueError:
 60            pass
 61    return gaze2dx_filtered, gaze2dy_filtered, time_stamp_filtered

 64def premier_indice(lst, value):
 65    """
 66    Trouve l'index de la première valeur ≥ value.\n
 67        • Entrées : Une liste triée (lst), une valeur seuil (value). \n
 68        • Logique : Parcourt la liste du début à la fin. \n
 69                    Retourne l'index du premier élément qui est supérieur ou égal à la valeur seuil. \n
 70        • Sorties : Un entier (index).
 71    """
 72
 73    for i, v in enumerate(lst):
 74        if float(v) >= value:
 75            return i
 76    return None

 79def dernier_indice(lst, value):
 80    """
 81    Trouve l'index de la denière valeur ≤ value.\n
 82        • Entrées : Une liste triée (lst), une valeur seuil (value). \n
 83        • Logique : Parcourt la liste du début à la fin. \n
 84                    Retourne l'index du denier élément qui est inférieur ou égal à la valeur seuil. \n
 85        • Sorties : Un entier (index).
 86    """
 87
 88    for i in reversed(range(len(lst))):
 89        if float(lst[i]) <= value:
 90            return i
 91    return None

 94def extrait_donnees_utiles (t1, t2, x_filtered, y_filtered, time_filtered) :
 95    """
 96    Cette fonction gère la synchronisation entre le temps absolu du fichier TSV et l'intervalle demandé par l'utilisateur (t1 à t2).\n
 97        • Entrées : Temps début (t1), temps fin (t2), listes filtrées (x,y,t).\n
 98        • Logique : Appelle premier_indice pour t1 et dernier_indice pour t2. \n
 99                    Découpe les listes Python entre ces deux bornes.\n
100        • Sorties : Sous-listes extraites correspondant à l'intervalle temporel.
101    """
102
103    if not time_filtered:
104        return [], [], []
105
106    t1_precis = premier_indice(time_filtered,t1)
107    t2_precis = dernier_indice(time_filtered,t2)
108
109    if t1_precis is None or t2_precis is None or t1_precis > t2_precis:
110        return [], [], []
111
112    x_extrait = x_filtered [t1_precis:t2_precis+1]
113    y_extrait = y_filtered [t1_precis:t2_precis+1]
114    time_extrait = time_filtered [t1_precis:t2_precis+1]
115
116    return x_extrait, y_extrait, time_extrait

165        • Entrées : Chemin vidéo, t1, t2, dossier sortie, listes de données nettoyées.\n
166        • Logique : Définit une fonction interne make_frame(t). Pour chaque instant t, calcule l'index \n
167                    des données correspondant, dessine l'historique des points sur la frame avec un \n
168                    dégradé de couleur et renvoie l'image modifiée.\n

170    """
171
172    prendre_extrait (video, t1, t2, dossier_sortie)
173    chemin_brut = os.path.join(dossier_sortie, f"{dossier_sortie}.mp4")

176        print("ERREUR : Pas de données de regard sur cet intervalle.")
177        return

188
189    for i in range(total_frames):
190        t_current = i / fps
191
192        while csv_ptr + 1 < len(time_vals) and time_vals[csv_ptr + 1] <= t_current:
193            csv_ptr += 1
194
195        if len(time_vals) > 0 and t_current >= time_vals[0]:
196            donnees_csv.append({
197                "Frame_Index": i + 1,
198                "Video_Time": t_current,
199                "Gaze_Time": time_extrait[csv_ptr],
200                "Gaze2dX": x[csv_ptr],
201                "Gaze2dY": y[csv_ptr]
202            })
203        else:
204            donnees_csv.append({
205                "Frame_Index": i + 1,
206                "Video_Time": t_current,
207                "Gaze_Time": "",
208                "Gaze2dX": "",
209                "Gaze2dY": ""
210            })
211
212    nom_csv_final = os.path.join(dossier_sortie, f"{dossier_sortie}_données_intermédiaires.csv")
213    df_out = pd.DataFrame(donnees_csv)
214    df_out.to_csv(nom_csv_final, index=False, sep=',')
215
216    current_idx = [0]
217
218
219    def make_frame(t):
220        frame = clip.get_frame(t)
221        frame = cv2.cvtColor(frame, cv2.COLOR_RGB2BGR)
222
223        while current_idx[0] + 1 < len(time_vals) and time_vals[current_idx[0] + 1] <= t:
224            current_idx[0] += 1
225

233            alpha = i / len(points_to_draw)
234            current_color = (
235                int(trace_color[0] * alpha),
236                int(trace_color[1] * alpha),
237                int(trace_color[2] * alpha)
238            )
239            cv2.line(frame, points_to_draw[i-1], points_to_draw[i], current_color, epaisseur)
240
241        x_t, y_t = int(x[(current_idx[0])]), int(y[current_idx[0]])
242        cv2.circle(frame, (x_t, y_t), epaisseur, trace_color, -1)
243
244
245        frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
246        return frame
247
248    chemin_final = os.path.join(dossier_sortie, f"{dossier_sortie}_chemin.mp4")
249    trace_clip = VideoClip(make_frame, duration=clip.duration).with_fps(clip.fps)
250    final = CompositeVideoClip([clip, trace_clip])
251    final.write_videofile(chemin_final, fps=clip.fps, codec="libx264", logger=None)

253    logging.info(f" -> Vidéo finale prête : {chemin_final}")

256def generer_images_boucle(video, t1, t2, nb_frames, dossier_sortie, x_filtered, y_filtered, time_filtered, epaisseur, mode="parcours",
257                          show_bg=True,  history_frames=0, bg_color=(0, 0, 0),trace_color=(255, 255, 0)):
258    """
259    Produit une séquence d'images fixes basées sur un échantillonnage temporel régulier.\n
260        • Entrées : Vidéo, temps, nombre d'images, dossier de sortie, variables nettoyées,\n
261                    options (mode tracé/point, fond noir/image).\n
262        • Logique : Calcule N timestamps équidistants. Boucle sur dessus, extrait l'image correspondante\n
263                    (ou crée une image noire), dessine les données eye-tracking synchrones et sauvegarde.\n
264        • Sortie : Série de fichiers frame_XXX.png écrits sur le disque.
265    """
266
267    prendre_extrait(video, t1, t2, dossier_sortie)
268    chemin_temp = os.path.join(dossier_sortie, f"{dossier_sortie}.mp4")
269
270    clip = VideoFileClip(chemin_temp)
271    x, y, time_extrait = extrait_donnees_utiles(t1, t2, x_filtered, y_filtered, time_filtered)
272    time_vals = [tt - t1 for tt in time_extrait] if x else []
273    timestamps = np.linspace(0, max(0, clip.duration-0.1), nb_frames)
274    w, h = clip.size
275    data_indices = []
276    ptr = 0
277    donnees_csv = []
278
279    for t_frame in timestamps:
280        while ptr < len(time_vals) and time_vals[ptr] <= t_frame:
281            ptr += 1
282        data_indices.append(ptr - 1)
283
284    for i, t_frame in enumerate(timestamps):
285
286        if show_bg:
287            try:
288                frame = cv2.cvtColor(clip.get_frame(t_frame), cv2.COLOR_RGB2BGR)
289            except IOError:
290                frame = np.zeros((h, w, 3), dtype=np.uint8)
291                frame[:] = bg_color
292        else:
293            frame = np.zeros((h, w, 3), dtype=np.uint8)
294            frame[:] = bg_color
295
296        idx_end_data = data_indices[i]
297
298        if idx_end_data >= 0:
299            donnees_csv.append({
300                "Num_Image": i + 1,
301                "Time": time_extrait[idx_end_data],
302                "Gaze2dX": x[idx_end_data],
303                "Gaze2dY": y[idx_end_data]
304            })
305        else:
306            donnees_csv.append({
307                "Num_Image": i + 1,
308                "Time": t1 + t_frame,
309                "Gaze2dX": "",
310                "Gaze2dY": ""
311            })
312
313        if idx_end_data >= 0:
314            if mode == "parcours":
315                idx_start_data = 0
316                if history_frames > 0:
317                    prev_img_idx = max(0, i - history_frames)
318                    idx_start_data = data_indices[prev_img_idx]
319
320                    idx_start_data = max(0, idx_start_data)
321
322                pts = [(int(x[j]), int(y[j])) for j in range(idx_start_data, idx_end_data + 1)]
323                for j in range(1, len(pts)):
324                    alpha = j / len(pts)
325                    current_color = (
326                        int(trace_color[0] * alpha),
327                        int(trace_color[1] * alpha),
328                        int(trace_color[2] * alpha)
329                    )
330                    cv2.line(frame, pts[j-1], pts[j], current_color, epaisseur)
331
332                if len(pts) > 0:
333                    cv2.circle(frame, pts[-1], epaisseur, trace_color, -1)
334            else:
335                cv2.circle(frame, (int(x[idx_end_data]), int(y[idx_end_data])), epaisseur, trace_color, -1)
336
337        cv2.imwrite(os.path.join(dossier_sortie, f"frame_{i+1:03d}.png"), frame)
338        logging.debug(f"      Image {i+1}/{nb_frames} OK")
339
340    nom_csv_final = os.path.join(dossier_sortie, f"{dossier_sortie}_données_intermédiaires.csv")
341    df_out = pd.DataFrame(donnees_csv)
342    df_out.to_csv(nom_csv_final, index=False, sep=',')
343    logging.info(f" -> CSV des images généré : {nom_csv_final}")
344
345    clip.close()
